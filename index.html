Favorite Place to Get Food: Chen's Buffet

Chen's Buffet is a Chinese restaurant in my hometown of Paola, Kansas. It's located just outside of town, next to our Dollar General and our hair salon. It rivals another Chinese restaurant in our town, Spring of China, but I prefer Chen's Buffets' selection and crab rangoon over anything Spring of China has.

Lowest Common Ancestor - Binary Lifting

Given a binary tree, the task is to find the Lowest Common Ancestor of the given two nodes in the tree. 

                
                    int n, l;
                    vector> adj;
                    
                    int timer;
                    vector tin, tout;
                    vector> up;
                    
                    void dfs(int v, int p)
                    {
                        tin[v] = ++timer;
                        up[v][0] = p;
                        for (int i = 1; i <= l; ++i)
                            up[v][i] = up[up[v][i-1]][i-1];
                    
                        for (int u : adj[v]) {
                            if (u != p)
                                dfs(u, v);
                        }
                    
                        tout[v] = ++timer;
                    }
                    
                    bool is_ancestor(int u, int v)
                    {
                        return tin[u] <= tin[v] && tout[u] >= tout[v];
                    }
                    
                    int lca(int u, int v)
                    {
                        if (is_ancestor(u, v))
                            return u;
                        if (is_ancestor(v, u))
                            return v;
                        for (int i = l; i >= 0; --i) {
                            if (!is_ancestor(up[u][i], v))
                                u = up[u][i];
                        }
                        return up[u][0];
                    }
                    
                    void preprocess(int root) {
                        tin.resize(n);
                        tout.resize(n);
                        timer = 0;
                        l = ceil(log2(n));
                        up.assign(n, vector(l + 1));
                        dfs(root, root);
                    }
